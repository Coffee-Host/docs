import { Callout } from "fumadocs-ui/components/callout";

import { Step, Steps } from "fumadocs-ui/components/steps";

<Steps>
	<Step>
	    登入客戶中心。
        ![圖片](/images/store/account/view-credit/login.png)
    </Step>
	<Step>
	    點擊右上角，進入 「我的詳情」。 
        ![圖片](/images/store/account/edit-account/myinfo.png)
    </Step>
    <Step>
        在此更改您的「個人資料」和「帳單地址」。
        ![圖片](/images/store/account/edit-account/about.png)
    </Step>
    <Step>
        修改完成後勾選「確認填寫真實資料 」。
        ![圖片](/images/store/account/edit-account/check.png)
    </Step>
    <Step>
        選擇「儲存變更 」。
        ![圖片](/images/store/account/edit-account/save.png)
    </Step>
</Steps>
