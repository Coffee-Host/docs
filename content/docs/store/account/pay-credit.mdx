---
title: 使用帳號餘額支付帳單
description: 使用帳號餘額，快速支付帳單。
---

import { Callout } from "fumadocs-ui/components/callout";

<Callout title="成功加值後，不接受任何理由退款或變成等值現金。" />

import { Step, Steps } from "fumadocs-ui/components/steps";

<Steps>
	<Step>
		登入客戶中心。 ![圖片](/images/store/account/pay-credit/login.png)
	</Step>
	<Step>
		選單 「帳務」，點擊 「我的帳單」。
		![圖片](/images/store/account/pay-credit/menu.png)
	</Step>
	<Step>
		點擊要支付的帳單。 ![圖片](/images/store/account/pay-credit/choose.png)
	</Step>
	<Step>
		更換付款方式到 「超大咖帳戶餘額」。
		![圖片](/images/store/account/pay-credit/payment.png)
	</Step>
	<Step>
		滑到下方，點擊 「使用餘額支付」。
		![圖片](/images/store/account/pay-credit/pay.png)
	</Step>
	<Step>
		稍待片刻後，帳單就會支付成功了。
		![圖片](/images/store/account/pay-credit/done.png)
	</Step>
</Steps>
